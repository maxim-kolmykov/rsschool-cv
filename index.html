    <div id="clock"></div>

        function getTime() {
            const date = new Date();
            const seconds = date.getSeconds();
            const minutes = date.getMinutes();
            const hours = date.getHours();
            clock.textContent =
            `${hours < 10 ? `0${hours}` : hours}:
             ${minutes < 10 ? `0${minutes}` : minutes}:
            ${seconds < 10 ? `0${seconds}` : seconds}`;
        }
        function init() {
            getTime();
            setInterval(getTime, 1000);
        }
        init();
    </script>
